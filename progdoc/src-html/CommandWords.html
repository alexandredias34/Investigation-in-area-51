002/**
003 *CommandWords définit quelles sont les commandes valides et comment les afficher
004 */
005public class CommandWords
006{
007    // a constant array that will hold all valid command words
008    private final String[] aValidCommands;
009
010    /**
011     * Constructeur, initialise les mots de commande valides
012     */
013    public CommandWords()
014    {
015        this.aValidCommands = new String[9];
016        this.aValidCommands[0] = "go";
017        this.aValidCommands[1] = "help";
018        this.aValidCommands[2] = "quit";
019        this.aValidCommands[3] = "look";
020        this.aValidCommands[4] = "picture";
021        this.aValidCommands[5] = "back";
022        this.aValidCommands[6] = "test"; 
023        this.aValidCommands[7] = "take";
024        this.aValidCommands[8] = "drop";
025    } // CommandWords()
026
027    /**
028     * Vérifie si la String passée en paramètre constitue un mot de commande valide.
029     * @return true si la String est un mot de commande valide, false si ce n'en est pas un
030     */
031    public boolean isCommand( final String pString )
032    {
033        for ( int i=0; i<this.aValidCommands.length; i++ )
034        {
035            if ( this.aValidCommands[i].equals( pString ) )
036                return true;
037        } // for
038        // if we get here, the string was not found in the commands
039        return false;
040    } // isCommand()
041    /**
042     * Affiche toutes les commandes valides (utilisant System.out)
043     * @return la liste des commandes valides
044     */
045    public String getCommandList()
046    {
047        String vCmdList= "";
048        for(String vCommand : aValidCommands) 
049        {
050            vCmdList  =vCmdList + vCommand + " ";
051        }
052        return vCmdList;
053    }//getCommandList()
054} // CommandWords
