001import java.util.HashMap;
002import java.util.Set;
003/**
004 * This class is part of the "World of Zuul" application.
005 * "World of Zuul" is a very simple, text based adventure game.
006 *
007 * @author Espinasse Baptiste
008 * @version 2017.12.16
009 */
010public class ItemList
011{
012    private HashMap<String, Item> aInventory;
013    /**
014     * Constructeur d'objets de classe ItemList
015     */
016    public ItemList()
017    {
018        this.aInventory = new HashMap<>();
019    }//ItemList()
020
021    /**
022     *Cette procédure a pour but d'afficher tous les items contenus
023    dans l'inventaire de la Room ou du Player
024     *@return Un String de tous les items contenu dans l'inventaire
025     */
026    public String getItemString(){
027        if (!this.aInventory.isEmpty())
028        {
029            String vItemString = new String("");
030            Set<String> keys = this.aInventory.keySet();
031
032            for (String vKey : keys)
033            {
034                vItemString += "\n" + this.aInventory.get(vKey).getLongDescription();
035            }
036            return vItemString;
037        }
038        else 
039        {return "";}
040    }//getItemString()
041    /**
042     *Procédure permettant d'ajouter un item à l'inventaire 
043     *@param pStringItem est une String correspondant au nom de l'Item à ajouter dans l'inventaire
044     *@param pItem est l'Item à ajouter dans l'inventaire
045     */
046    public void takeItem(final String pStringItem, final Item pItem){this.aInventory.put(pStringItem, pItem);}//takeItem(..) 
047    /**
048     *Procédure permettant de retirer un Item de l'inventaire
049     *@param pStringItem est une String contenant le nom de l'Item à retirer de l'inventaire
050     */
051    public void dropItem(final String pStringItem){this.aInventory.remove(pStringItem);}//dropItem(.)
052    /**
053     *Accesseur d'un item à partir du nom de l'Item
054     *@param pItem est une String correspondant au nom de l'item que l'on souhaite obtenir dans l'inventaire
055     */
056    public Item getItem(final String pItem){return this.aInventory.get(pItem);}//getItem(.)
057    /**
058     * Booléen représentant la vérification de la présence d'un item dans l'inventaire
059     * @param pKey est une String correspondant à la clé de l'Item
060     */
061    public boolean hasItem(final String pKey)
062    {
063        return this.aInventory.containsKey(pKey);
064    }//hasItem(.)
065    /**
066     * Procédure permettant l'ajout de l'Item à l'inventaire, en connaissant sont nom.
067     * @param pName est le nom de l'Item
068     * @param pItem est l'Item
069     */
070    public void addItem (final String pName, final Item pItem)
071    {
072        this.aInventory.put(pName,pItem);
073    }//addItem(..)
074    /**
075     * Procédure permettant l'ajout de l'Item à l'inventaire, en sollicitant un accesseur
076     * @param pItem est l'Item
077     */
078    public void addItem(final Item pItem)
079    {
080        this.aInventory.put(pItem.getItemName(), pItem);
081    }//addItem(.)
082    /**
083     * Procédure permettant l'ajout de l'Item à l'inventaire (généralement d'une Room), permettant de définir nom, description et poids de l'item.
084     *@param pItemName est le nom de l'Item
085     *@param pItemDesc est la description de l'Item
086     *@param pItemWeight est le poids de l'Item
087     */
088    public void addItem(final String pItemName, final String pItemDesc, final int pItemWeight)
089    {
090        this.aInventory.put(pItemName, new Item(pItemName, pItemDesc,pItemWeight));
091    } //addItem(...)
092    /**
093     * Procédure permettant de retirer l'Item de l'inventaire
094     * @param pItem est le nom de l'Item.
095     */
096    public void removeItem (final String pItem)
097    {
098        this.aInventory.remove(pItem);
099    }
100
101}
